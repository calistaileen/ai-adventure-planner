@app.post("/api/user_trips")
def create_user_trip():
    conn = get_db()
    uid  = current_user_id()      # 可能為 None（未登入）
    tok  = _uid_token()           # 供未登入者關聯資料

    data  = request.get_json(silent=True) or {}
    title = (data.get("title") or "Untitled Trip").strip()

    # 把前端的行程物件打包進 notes（字串 JSON）
    # 盡量保留 stops/points/days，之後 saved.html 能解析
    payload = {
        "title": title,
        "notes": data.get("notes"),
        "stops": data.get("stops") or data.get("points") or [],
        "days":  data.get("days")  or []
    }
    notes_str = json.dumps(payload, ensure_ascii=False)

    try:
        with conn.cursor() as cur:
            cur.execute("""
                INSERT INTO user_trips
                  (user_id, user_token, title, notes, base_trip_id, created_at, updated_at)
                VALUES
                  (%s, %s, %s, %s, %s, NOW(), NOW())
            """, (uid, tok, title, notes_str, None))
            new_id = cur.lastrowid
        return jsonify({"ok": True, "user_trip_id": int(new_id)})
    except Exception as e:
        return jsonify({"ok": False, "error": str(e)}), 500


# List my trips（同時包含：登入者 user_id 與 X-UID token）
@app.get("/api/user_trips")
def list_user_trips():
    conn = get_db()
    uid  = current_user_id()  # 可能為 None
    tok  = _uid_token()

    try:
        with conn.cursor() as cur:
            cur.execute("""
                SELECT id, user_id, user_token, title, notes, created_at
                FROM user_trips
                WHERE
                  (%s IS NOT NULL AND user_id = %s)
                  OR
                  (%s IS NOT NULL AND user_token = %s)
                ORDER BY created_at DESC, id DESC
            """, (uid, uid, tok, tok))
            rows = cur.fetchall()
        return jsonify({"ok": True, "items": rows})
    except Exception as e:
        return jsonify({"ok": False, "error": str(e)}), 500

@app.get("/api/user_trips/<int:trip_id>")
def get_user_trip(trip_id):
    conn = get_db()
    uid  = current_user_id()
    tok  = _uid_token()

    try:
        with conn.cursor() as cur:
            cur.execute("""
                SELECT id, user_id, user_token, title, notes, base_trip_id, created_at, updated_at
                FROM user_trips
                WHERE id = %s
                  AND (
                    (%s IS NOT NULL AND user_id = %s) OR
                    (%s IS NOT NULL AND user_token = %s)
                  )
                LIMIT 1
            """, (trip_id, uid, uid, tok, tok))
            row = cur.fetchone()
            if not row:
                return jsonify({"ok": False, "error": "not_found"}), 404
        return jsonify({"ok": True, "trip": row})
    except Exception as e:
        return jsonify({"ok": False, "error": str(e)}), 500

# ===== 建立或更新（同名視為更新）=====
@app.route("/api/user_trips", methods=["POST"])
def create_or_update_user_trip():
    from flask import request, jsonify
    import json

    body = request.get_json(silent=True) or {}
    trip = body.get("trip") or body
    title = (trip.get("title") or "").strip() or "Untitled"

    # 重要：把整包行程存到 notes
    trip_json = json.dumps(trip, ensure_ascii=False)

    # 同時拿到 user_id 與 user_token
    uid, utok = current_actor()

    conn = get_db()
    try:
        with conn.cursor() as cur:
            # 先找「同名、同人」：對已登入（user_id>0）用 user_id，比對；對匿名用 user_token 比對
            cur.execute(
                """
                SELECT id FROM user_trips
                WHERE title=%s AND (
                  (user_id=%s AND %s<>0) OR
                  (user_token=%s AND %s=0)
                )
                LIMIT 1
                """,
                (title, uid, uid, utok, uid),
            )
            row = cur.fetchone()

            if row:
                # 更新
                cur.execute(
                    """
                    UPDATE user_trips
                    SET notes=%s, updated_at=NOW()
                    WHERE id=%s
                    """,
                    (trip_json, row["id"]),
                )
                conn.commit()
                return jsonify({"ok": True, "trip_id": row["id"], "mode": "update"})
            else:
                # 新增（同時寫入 user_token，避免 NOT NULL 或未來查詢不到）
                cur.execute(
                    """
                    INSERT INTO user_trips (user_id, user_token, title, notes, base_trip_id, created_at, updated_at)
                    VALUES (%s, %s, %s, %s, %s, NOW(), NOW())
                    """,
                    (uid, utok, title, trip_json, None),
                )
                new_id = cur.lastrowid
                conn.commit()
                return jsonify({"ok": True, "trip_id": new_id, "mode": "insert"})
    except Exception as e:
        conn.rollback()
        # 臨時把實際錯誤也回給前端，方便你排錯（穩定後可改成統一訊息）
        return jsonify({"ok": False, "error": str(e)}), 500
    finally:
        conn.close()

    if (!Array.isArray(stops) || stops.length < 2) return;

    // 只補缺：若每一段都已有 leg（有 mode 或手動時長/距離），就直接跳過
    if (onlyMissing) {
      const hasGap = stops.some((s, i) => {
        if (i === 0) return false; // 第一個點沒有入段
        const leg = s.leg;
        return !(leg && (leg.mode || leg.manualDurationMin || leg.manualDistanceKm));
      });
      if (!hasGap) {
        console.log('[applyAiLegs] 已無缺少的交通段，跳過 AI 計算');
        return;
      }
    }

    const payload = {
      stops: stops.map(s => ({
        name: s.name || '',
        lat:  s.position?.lat,
        lng:  s.position?.lng
      }))
    };

    console.log('[ai/legs] payload stops =', payload.stops);

    const res  = await fetch(apiUrl('ai/legs'), {
      method: 'POST',
      headers: { 'Content-Type':'application/json' },
      body: JSON.stringify(payload)
    });

    const data = await res.json().catch(()=> ({}));

    if (data?.ok && Array.isArray(data.legs)) {
      data.legs.forEach(leg => {
        const i = leg.index;            // 1..N-1，表示 A->B 的 B 是第 i 個
        if (!stops[i]) return;

        // 只補缺：如果原本就有手動/明確的 leg，就不要覆蓋
        if (onlyMissing && (stops[i].leg && (stops[i].leg.mode || stops[i].leg.manualDurationMin || stops[i].leg.manualDistanceKm))) {
          return;
        }

        stops[i].leg = Object.assign(stops[i].leg || {}, leg);
      });
    }
  }
